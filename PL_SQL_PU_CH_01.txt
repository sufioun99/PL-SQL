--Anonymous Block
==================

DECLARE				--Declarative section (optional)
	CURSOR c_emp IS
        	SELECT * FROM employees
        	WHERE employee_id > 300;
	v_lname VARCHAR2(15);
	r_emp employees%ROWTYPE;
BEGIN				--Executable section (mandatory)
	SELECT * INTO r_emp FROM employees
	WHERE employee_id > 300;
	
	SELECT last_name INTO v_lname
	FROM employees
	WHERE first_name='John';
	DBMS_OUTPUT.PUT_LINE ('John''s last name is :' ||v_lname);

	CLOSE c_emp;
EXCEPTION			--Exception section (optional)
	WHEN NO_DATA_FOUND THEN
	DBMS_OUTPUT.PUT_LINE (' Select statement returned no data.');

	WHEN TOO_MANY_ROWS THEN
	DBMS_OUTPUT.PUT_LINE (' Select statement retrieved multiple rows. Consider using a cursor.');

	WHEN OTHERS THEN
	DBMS_OUTPUT.PUT_LINE (' For all other errors.');
END;				--End of block (mandatory)
/


--Using IN Parameters
=====================

CREATE OR REPLACE PROCEDURE raise_salary
  	(p_id IN employees.employee_id%type,
 	p_percent IN NUMBER)
  	IS
BEGIN
  	UPDATE employees
  	SET salary = salary * (1 + p_percent/100)
  	WHERE employee_id = p_id;
END raise_salary;
/

SELECT salary FROM employees WHERE employee_id = 176;

EXECUTE raise_salary(176,10);

SELECT salary FROM employees WHERE employee_id = 176;

--Using OUT Parameters
======================

CREATE OR REPLACE PROCEDURE query_emp
  	(p_id IN employees.employee_id%TYPE,
  	p_name OUT employees.last_name%TYPE,
  	p_salary OUT employees.salary%TYPE)
  	IS
BEGIN
  	SELECT last_name, salary INTO p_name, p_salary
  	FROM employees
  	WHERE employee_id = p_id;
END query_emp;
/

SELECT salary FROM employees WHERE employee_id = 171;

DECLARE
  	v_emp_name employeeS.last_name%TYPE;
  	v_emp_sal employees.salary%TYPE;
BEGIN
  	query_emp(171, v_emp_name, v_emp_sal);
	DBMS_OUTPUT.PUT_LINE(v_emp_name||' earns '|| to_char(v_emp_sal, '$999,999.00'));
END;
/



--Using IN OUT Parameters
==========================

CREATE OR REPLACE PROCEDURE format_phone
  	(p_phone_no IN OUT VARCHAR2) IS
BEGIN
 	p_phone_no :='(' || SUBSTR(p_phone_no,1,3) ||
  	')' || SUBSTR(p_phone_no,4,3) ||
  	'-' || SUBSTR(p_phone_no,7);
END format_phone;
/



VARIABLE phone_no VARCHAR2(15)

EXECUTE :phone_no := '80006330575'


PRINT phone_no


EXECUTE format_phone(:phone_no)


PRINT phone_no

--Passing actual parameters 
===========================

CREATE OR REPLACE PROCEDURE add_dept(
  	p_name IN departments.department_name%TYPE,
  	p_loc IN departments.location_id%TYPE)
  	IS
BEGIN
  	INSERT INTO departments(department_id,
  	department_name, location_id)
  	VALUES (departments_seq.NEXTVAL, p_name, p_loc);
END add_dept;
/


--Passing parameters using the positional notation

EXECUTE add_dept ('TRAINING', 2500)

--Passing parameters using the named natation

EXECUTE add_dept (p_loc=>2400, p_name=>'EDUCATION')

SELECT department_id, department_name, location_id FROM departments;

--Calling Procedures
====================
CREATE OR REPLACE PROCEDURE process_employees
	IS
	CURSOR cur_emp_cursor IS
		SELECT employee_id
		FROM employees;
BEGIN
	FOR emp_rec IN cur_emp_cursor
	LOOP
		raise_salary(emp_rec.employee_id, 10);
	END LOOP;
	COMMIT;
END process_employees;
/

--Handled Exceptions
====================
CREATE PROCEDURE add_department(
	p_name VARCHAR2, p_mgr NUMBER, p_loc NUMBER) IS
BEGIN
	INSERT INTO DEPARTMENTS (department_id, department_name, manager_id, location_id)
			VALUES (DEPARTMENTS_SEQ.NEXTVAL, p_name, p_mgr, p_loc);
	DBMS_OUTPUT.PUT_LINE('Added Dept: '|| p_name);
EXCEPTION
	WHEN OTHERS THEN
	DBMS_OUTPUT.PUT_LINE('Err: adding dept: '|| p_name);
END;
/

CREATE PROCEDURE create_departments IS
BEGIN
	add_department('Media', 100, 1800);
	add_department('Editing', 99, 1800);
	add_department('Advertising', 101, 1800);
END;
/
